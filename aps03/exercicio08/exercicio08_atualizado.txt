Algoritmo "cadastroConsultaNome"
Var
    contador, continuar: inteiro
    nome: vetor[1..10] de caractere

// Função para verificar se existe um nome cadastrado na posição digitada
funcao buscarPosicao(posicao: inteiro): caractere
var resultado: caractere
inicio
	se (nome[posicao]) entao
        resultado <- "O nome cadastrado na posição ", posicao, " é ", nome[posicao]
    senao
        resultado <- "Não encontramos nenhum nome cadastrado nessa posição"
    fimse

    retorne resultado
fimfuncao

// Procedimento do menu
procedimento menu()
var opcao: inteiro
inicio
    escreval("+----- ### MENU ### -----+")
    escreval("| 1 – Cadastrar nomes    |")
    escreval("| 2 – Buscar por posição |")
    escreval("| 3 – Buscar por nome    |")
    escreval("| 4 – Imprimir nomes     |")
    escreval("+------------------------+")

    escreva("Digite uma opção do menu: ")
    leia(opcao)

    escolha opcao
        caso 1
            menuCadastro()
        caso 2
            menuBuscaPosicao()
        caso 3
            menuBuscaNome()
        caso 4
            menuImprimir()
        outrocaso
            limpatela
            escreval()
            escreval("Digite um número entre 1 e 4.")
            menu()
    fimescolha
fimprocedimento

// Procedimento de cadastro dos usuários
procedimento menuCadastro()
inicio
    para contador de 1 ate 10 faca
        escreva("Digite nome #", contador, ": ")
        leia(nome[contador])
    fimpara

    limpatela
    menu()
fimprocedimento

// Procedimento de busca por posição
procedimento menuBuscaPosicao()
var numero: inteiro
inicio
    repita
        escreva("Digite uma posição para verificar o nome cadastrado: ")
        leia(numero)
    ate (numero < 1)
    
    escreval(buscarPosicao(numero))

    repita
        escreval()
        escreva("Deseja continuar? (1 - sim, 2 - não): ")
        leia(continuar)
    ate (continuar > 2 e continuar < 1)

    se (continuar = 1) entao
        limpatela
        menu()
    fimse
fimprocedimento

// Procedimento de busca por nome
procedimento menuBuscaNome()
var
    verificar: inteiro
    pesquisa: caractere
inicio
    escreva("Digite um nome para verificar se está cadastrado: ")
    leia(pesquisa)

    verificar <- 0

    para contador de 1 ate 10 faca
        se (pesquisa = nome[contador]) entao
            escreval("O nome ", pesquisa, " está cadastrado na posição ", contador)
        senao
            verificar++;
        fimse
    fimpara

    se (verificar >= 10) entao
        escreval("O nome ", pesquisa, " não foi encontrado")	
    fimse

    repita
        escreval()
        escreva("Deseja continuar? (1 - sim, 2 - não): ")
        leia(continuar)
    ate (continuar > 2 e continuar < 1)

    se (continuar = 1) entao
        limpatela
        menu()
    fimse
fimprocedimento

// Procedimento que exibe todos os usuários cadastrados na tela
procedimento menuImprimir()
inicio
    para contador de 1 ate 10 faca
        se (nome[contador] != "") entao
            escreval("Nome #", contador, ": ", nome[contador])
        senao
            se (contador = 10 e nome[contador] = "") entao
                escreval("Nenhum nome encontrado")
                pare
            fimse
        fimse
    fimpara

    repita
        escreval()
        escreva("Deseja continuar? (1 - sim, 2 - não): ")
        leia(continuar)
    ate (continuar > 2 e continuar < 1)

    se (continuar = 1) entao
        limpatela
        menu()
    fimse
fimprocedimento
